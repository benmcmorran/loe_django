# onetime thing: set old teams to inactive
for cont_id in range(500):
    teams = Team.objects.filter(team_continuity_id=cont_id)
    if teams.count() > 1:
        old_teams = teams.order_by('-teamrating__rating_date')[1:]
        for ot in old_teams:
            ot.is_active = False
            ot.save()
